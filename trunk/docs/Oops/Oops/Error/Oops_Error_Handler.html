Class for handling user errors

Usage:

  1. <?
  2.    require_once("Oops/Error/Handler");
  3.    $eh = new Oops_Error_Hander();
  4.  
  5.    // ... your code ...
  6.  
  7.    trigger_error("Some error occured", E_USER_ERROR);
  8.    trigger_error("Warning here", E_USER_WARNING);
  9.    trigger_error("Sending a notice", E_USER_NOTICE);
  10.  
  11.    //... your code ...
  12.  
  13.    if($eh->isClear()) {
  14.       //there were no errors/warnings/notices in your code
  15.    }
  16.  
  17.    if($eh->isError()) {
  18.       //there were user errors
  19.       Oops_Debug::Dump($eh->getErrors(),"Errors occured"));
  20.    }
  21.  
  22.    if($eg->isWarning()) {
  23.       //there were warnings
  24.       Oops_Debug::Dump($eh->getWarnings(),"Warnings occured"));
  25.    }
  26.  
  27.    if($eg->isNotice()) {
  28.       //there were notices
  29.       Oops_Debug::Dump($eh->getNotices(),"Notices sent"));
  30.    }
  31.  
  32.    //when using PHP4 call destructor manually
  33.    $eh->destruct();
  34.    unset($eh);
  35.  ?>
